-- DESIGN: Hygienic procedural MoonLift codegen in .mlua
-- ========================================================
--
-- GOAL: Lua loops produce MoonLift. MoonLift compiles to native.
--       Zero manual string concatenation of MoonLift syntax in user code.
--
-- The escape hatch ("source" quote kind) remains for rare cases,
-- but the primary path is ASDL values that know how to render themselves.
--
-- ═══════════════════════════════════════════════════════════
--  LAYER 1: New HostValueKind variants (schema/host.lua)
-- ═══════════════════════════════════════════════════════════
--
-- Add to HostValueKind sum:
--
--   HostValueStmt         -- a single MoonTree.Stmt
--   HostValueExpr         -- a single MoonTree.Expr
--   HostValueBlock        -- a single MoonTree.ControlBlock / EntryControlBlock
--   HostValueList         -- ordered collection of any spliceable value
--
-- ═══════════════════════════════════════════════════════════
--  LAYER 2: New SpliceExpectation variants (schema/host.lua)
-- ═══════════════════════════════════════════════════════════
--
-- Add to SpliceExpectation sum:
--
--   SpliceStmt            -- statement position in function/block/if body
--   SpliceStmtList        -- list of statements (case body, then body, etc.)
--   SpliceSwitchCase      -- switch case arm (key + body pair)
--   SpliceDecl            -- module item position (func, struct, etc.)
--
-- ═══════════════════════════════════════════════════════════
--  LAYER 3: moon.list() — the collection primitive
-- ═══════════════════════════════════════════════════════════
--
-- Lua API:
--
--   local L = moon.list()
--   L:push(value)          -- append a spliceable value
--   L:extend(other_list)   -- concatenate another list
--   L:len()                -- number of elements
--   L:get(i)               -- element at index (1-based)
--
-- Properties:
--   - moonlift_quote_kind = "list"
--   - moonlift_splice_source() iterates elements, calls splice_to_source
--     on each, joins with "\n"
--   - Elements can be: MoonTree.Stmt, MoonTree.Expr, MoonTree.ControlBlock,
--     RegionFragValue, ExprFragValue, SourceValue, plain numbers/strings,
--     or nested lists
--
-- ═══════════════════════════════════════════════════════════
--  LAYER 4: moonlift_splice_source() on MoonTree nodes
-- ═══════════════════════════════════════════════════════════
--
-- Each ASDL node that can appear in a list gets a method that
-- renders it to valid MoonLift source text. The pretty-printer
-- lives on the ASDL type, not in user code.
--
-- KEY NODES (MoonTree.*):
--
--   StmtEmit{ frag, args, fill_map }
--     → "emit @{frag_name}(arg1, arg2; exit1 = block1, exit2 = block2)"
--
--   StmtJump{ label, args }
--     → "jump label_name(arg = val, ...)"
--
--   StmtIf{ cond, then_body, else_body }
--     → "if cond then\n    stmts...\nend"
--
--   StmtSwitch{ value, cases, default }
--     → "switch val do\n    case KEY then\n        stmts...\ndefault then\n    stmts...\nend"
--
--   StmtLet{ name, ty, init }
--     → "let name: T = init"
--
--   StmtVar{ name, ty, init }
--     → "var name: T = init"
--
--   StmtAssign{ place, value }
--     → "place = value"
--
--   StmtYield{ value }
--     → "yield" / "yield value"
--
--   StmtReturn{ value }
--     → "return" / "return value"
--
--   StmtExpr{ expr }
--     → "expr"
--
--   ExprInt{ raw }
--     → "42"
--
--   ExprFloat{ raw }
--     → "3.14"
--
--   ExprName{ name }
--     → "x"
--
--   ExprBinary{ op, left, right }
--     → "left OP right"
--
--   ExprUnary{ op, operand }
--     → "OP operand"
--
--   ExprCall{ callee, args }
--     → "callee(arg1, arg2)"
--
--   ExprIndex{ base, index }
--     → "base[index]"
--
--   ExprField{ base, field }
--     → "base.field"
--
--   ExprDeref{ base }
--     → "*base"
--
--   ExprAddrOf{ place }
--     → "&place"
--
--   ExprAs{ target, source }
--     → "as(target, source)"
--
--   ExprSelect{ cond, then_expr, else_expr }
--     → "select(cond, then_expr, else_expr)"
--
--   ControlBlock{ label, params, body }
--     → "block label(params)\n    stmts...\nend"/"jump ..."
--
--   EntryControlBlock{ label, params, body }
--     → "entry label(params = init)\n    stmts...\nend"/"jump ..."
--
--   SwitchStmtArm{ key, body }
--     → "case KEY then\n    stmts..."
--
-- ═══════════════════════════════════════════════════════════
--  LAYER 5: Builder API for constructing nodes (lua/moonlift/ast.lua)
-- ═══════════════════════════════════════════════════════════
--
-- The low-level ast builder (already exists) exposes constructors
-- for each MoonTree node type. We need convenient wrappers:
--
--   moon.stmt_emit(frag, args, exits)      → MoonTree.StmtEmit
--   moon.stmt_jump(label, args)            → MoonTree.StmtJump
--   moon.stmt_if(cond, then_body, else)    → MoonTree.StmtIf
--   moon.stmt_switch(val, cases, def)      → MoonTree.StmtSwitch
--   moon.stmt_let(name, T, init)           → MoonTree.StmtLet
--   moon.stmt_return(expr)                 → MoonTree.StmtReturn
--   moon.stmt_yield(expr)                  → MoonTree.StmtYield
--   moon.expr_int(val)                     → MoonTree.ExprInt
--   moon.expr_binary(op, l, r)             → MoonTree.ExprBinary
--   moon.expr_name(name)                   → MoonTree.ExprName
--   moon.case_arm(key, body_stmts)         → MoonTree.SwitchStmtArm
--   moon.block_def(label, params, stmts)   → MoonTree.ControlBlock
--
-- These return ASDL values with the appropriate moonlift_quote_kind
-- metatable set. They carry moonlift_splice_source() that renders
-- valid MoonLift source.
--
-- ═══════════════════════════════════════════════════════════
--  LAYER 6: Expected splice inference (mlua_host_model.lua)
-- ═══════════════════════════════════════════════════════════
--
-- The expected_splice() function inspects the source context
-- before @{...} and determines what kind of value is expected.
--
-- Current logic:
--   if prefix ends with "emit"     → SpliceEmit
--   if line ends with ":" pattern  → SpliceType
--   default                        → SpliceExpr
--
-- New logic:
--   if inside "switch ... do" and before "default"/"end"
--     → SpliceSwitchCase
--   if after "then", "else", newline in statement position
--     → SpliceStmt
--   (keep existing Emit/Type/Expr inference)
--

-- ═══════════════════════════════════════════════════════════
--  EXAMPLE: Trie lexer with hygienic codegen
-- ═══════════════════════════════════════════════════════════
--
-- BEFORE (string codegen):
--   local switch_str = string.format("case %d then emit @{%s}(...)", ...)
--   return region ... entry ... switch c do @{switch_str} ... end
--
-- AFTER (hygienic):
--   local cases = moon.list()
--   for byte, child in sorted_pairs(children) do
--       cases:push(moon.case_arm(
--           moon.expr_int(byte),
--           moon.list():push(
--               moon.stmt_emit(child,
--                   {moon.expr_name("p"), moon.expr_name("n"), moon.expr_name("pos")},
--                   {ok = "ok", fail = "fail"}
--               )
--           )
--       ))
--   end
--   return region trie_node(p: ptr(u8), n: i32, pos: i32;
--       ok: cont(next: i32, token: i32), fail: cont(at: i32))
--   entry start()
--       let offset: i32 = pos + @{depth}
--       if offset >= n then jump fail(at = pos) end
--       let c: i32 = as(i32, p[offset])
--       switch c do
--           @{cases}
--       default then
--           jump fail(at = pos)
--       end
--   end
--   end
--
-- The @{cases} splice triggers:
--   1. cases:moonlift_splice_source()
--   2. Iterates list elements (SwitchStmtArm values)
--   3. Each renders: "case 65 then\n    emit r_child_65(p, n, pos; ok = ok, fail = fail)"
--   4. Joined by "\n"
--   5. Inserted into the region source stream
--   6. Full source parsed by mlua_parse → MoonOpen.RegionFrag
--
-- Zero user-written MoonLift syntax in Lua strings.
-- All syntax knowledge lives in the pretty-printer.
